# Supabase 配置详细指南

## 1. 环境变量 (.env.local)

```env
# Supabase 项目 URL
NEXT_PUBLIC_SUPABASE_URL=https://你的项目.supabase.co

# Supabase 匿名密钥（公开，用于前端）
NEXT_PUBLIC_SUPABASE_ANON_KEY=eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9...
```

**获取位置：** Supabase Dashboard → Project Settings → API
- `URL`: Project URL
- `anon public`: anon (public) 密钥

---

## 2. Supabase 客户端 (lib/supabase.ts)

```typescript
import { createClient } from '@supabase/supabase-js'

// 从环境变量读取
const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL!
const supabaseKey = process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!

// 创建客户端实例
export const supabase = createClient(supabaseUrl, supabaseKey)
```

## 3. 数据库表结构

### 建表 SQL（在 Supabase SQL Editor 执行）

```sql
-- 创建 todos 表
CREATE TABLE todos (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
  title TEXT NOT NULL,
  completed BOOLEAN DEFAULT false NOT NULL
);

-- 启用行级安全策略 (RLS)
ALTER TABLE todos ENABLE ROW LEVEL SECURITY;

-- 允许匿名用户读取所有数据（开发阶段）
CREATE POLICY "Allow anonymous read" ON todos
  FOR SELECT USING (true);

-- 允许匿名用户插入数据
CREATE POLICY "Allow anonymous insert" ON todos
  FOR INSERT WITH CHECK (true);

-- 允许匿名用户更新数据
CREATE POLICY "Allow anonymous update" ON todos
  FOR UPDATE USING (true);

-- 允许匿名用户删除数据
CREATE POLICY "Allow anonymous delete" ON todos
  FOR DELETE USING (true);
```

## 4. 实时订阅配置

### 启用 Realtime（Supabase Dashboard）

1. 进入 Database → Replication
2. 找到 `todos` 表
3. 开启 `Insert`, `Update`, `Delete` 的实时推送

或者在 SQL Editor 执行：

```sql
-- 启用 todos 表的实时功能
ALTER PUBLICATION supabase_realtime ADD TABLE todos;
```

## 5. API 调用详解

### 获取所有任务

```typescript
const { data, error } = await supabase
  .from('todos')
  .select('*')
  .order('created_at', { ascending: false })

// data: Todo[]
// error: PostgrestError | null
```

### 添加任务

```typescript
const { data, error } = await supabase
  .from('todos')
  .insert([{ title: '买牛奶', completed: false }])
  .select()

// 返回插入的数据（含自动生成的 id）
```

### 更新任务状态

```typescript
const { error } = await supabase
  .from('todos')
  .update({ completed: true })
  .eq('id', 1)

// eq('id', 1) 表示只更新 id=1 的记录
```

### 删除任务

```typescript
const { error } = await supabase
  .from('todos')
  .delete()
  .eq('id', 1)
```

### 实时订阅（数据变化自动刷新）

```typescript
const subscription = supabase
  .channel('todos')
  .on(
    'postgres_changes',
    { 
      event: '*',        // * = 监听所有操作 (insert/update/delete)
      schema: 'public', 
      table: 'todos' 
    },
    (payload) => {
      console.log('数据变化:', payload)
      // payload.new = 新数据
      // payload.old = 旧数据
      // payload.eventType = 'INSERT' | 'UPDATE' | 'DELETE'
    }
  )
  .subscribe()

// 取消订阅
supabase.removeChannel(subscription)
```

## 6. 生产环境建议

### 更安全的 RLS 策略（登录用户）

```sql
-- 改为只允许登录用户操作
CREATE POLICY "Users can CRUD their own todos" ON todos
  FOR ALL
  USING (auth.uid() = user_id)
  WITH CHECK (auth.uid() = user_id);
```

需要添加 `user_id` 列关联到 `auth.users`。

### 添加 user_id 列

```sql
-- 添加用户关联列
ALTER TABLE todos ADD COLUMN user_id UUID REFERENCES auth.users(id);

-- 创建触发器：自动填充 user_id
CREATE OR REPLACE FUNCTION set_todos_user_id()
RETURNS TRIGGER AS $$
BEGIN
  NEW.user_id = auth.uid();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER set_todos_user_id_trigger
  BEFORE INSERT ON todos
  FOR EACH ROW
  EXECUTE FUNCTION set_todos_user_id();
```

## 7. 完整 lib/supabase.ts（增强版）

```typescript
import { createClient } from '@supabase/supabase-js'

const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL!
const supabaseKey = process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!

if (!supabaseUrl || !supabaseKey) {
  throw new Error('Missing Supabase environment variables')
}

export const supabase = createClient(supabaseUrl, supabaseKey, {
  auth: {
    persistSession: true,
    autoRefreshToken: true,
  },
  realtime: {
    params: {
      eventsPerSecond: 10,
    },
  },
})

// 类型定义
export interface Todo {
  id: number
  created_at: string
  title: string
  completed: boolean
  user_id?: string
}

// 辅助函数
export async function getTodos() {
  const { data, error } = await supabase
    .from('todos')
    .select('*')
    .order('created_at', { ascending: false })
  
  if (error) throw error
  return data as Todo[]
}

export async function createTodo(title: string) {
  const { data, error } = await supabase
    .from('todos')
    .insert([{ title, completed: false }])
    .select()
    .single()
  
  if (error) throw error
  return data as Todo
}

export async function updateTodo(id: number, completed: boolean) {
  const { error } = await supabase
    .from('todos')
    .update({ completed })
    .eq('id', id)
  
  if (error) throw error
}

export async function deleteTodo(id: number) {
  const { error } = await supabase
    .from('todos')
    .delete()
    .eq('id', id)
  
  if (error) throw error
}

export function subscribeToTodos(callback: (payload: any) => void) {
  return supabase
    .channel('todos')
    .on(
      'postgres_changes',
      { event: '*', schema: 'public', table: 'todos' },
      callback
    )
    .subscribe()
}
```
